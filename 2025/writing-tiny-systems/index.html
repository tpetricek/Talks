// A term  can be either atom (...),
// variable (...), or predicate (...).
type Term =
  | Atom of string
  | Variable of string
  | Predicate of string * Term list

// A clause is for example:
// 'mortal(X) :- human(X)'
// It consists of 'head :- body'
type Clause =
  { Head : Term
    Body : Term list }

// A program is a list of clauses
type Program = Clause list
